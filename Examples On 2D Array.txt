1- Write a program that take 2 
arrays (2D) from user and Print 
the summation in another 2D 
Array
                                                             Answer
#include <iostream> 
using namespace std;
int main() {
	int arr1[3][4]{}, arr2[3][4]{}, sum[3][4]{};
	cout << "Enter Values Of First Array\n";
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 4; j++) {
			cin >> arr1[i][j];
		}
	}
	cout << "================================================\n";
	cout << "Enter Values Of Second Array\n";
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 4; j++) {
			cin >> arr2[i][j];
		}
	}

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 4; j++) {
			sum[i][j] = arr1[i][j] + arr2[i][j];
		}
	}
	cout << "================================================\n";
	cout << "The Sumtion Of Two Arrays\n";
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 4; j++) {
			cout << sum[i][j] <<  " ";
		}
		cout << endl;
	}
	 
}

2- Write a program that Print the 
max of 2D Array.

                                                             Answer
#include <iostream>
using namespace std;
int main() {
int arr[3][3], max = INT_MIN;

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			cin >> arr[i][j];
			if (arr[i][j] > max) {
				max = arr[i][j];
			}
		}
	}

	cout << "Max = " << max << endl;
}

3- Write a program that Check if 
the matrix is identity or not
(identity matrix: all numbers must 
be zero and the diagonal is 1).
                                                                Answer

#include <iostream>
using namespace std;
int main() {
	int arr[3][3];
	bool flag = true;

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			cin >> arr[i][j];
		}
	}
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			if (i == j) {
				if (arr[i][j] != 1) {
					flag = false;
				}
			}
			else if (arr[i][j] != 0) {
				flag = false;
		}
		}
	}
	if (flag) {
		cout << "Identity Matrix\n";
	}
	else {
		cout << "Not Identity Matrix\n";
	}


}

4- Write a program that Search for 
value in 2D Array.
                                                           Answer
#include <iostream>
using namespace std;
int main() {
	int arr[3][3]{}, num{};
	bool flag = false;
	cout << "Enter Elements of Array\n";

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			cin >> arr[i][j];
		}
	}
	cout << "Enter Number To Search\n";
	cin >> num;
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
	 		if (arr[i][j] == num) {
				flag = true;
			}
		}
	}
	if (flag) {
		cout << num << " Is Founded\n";

	}
	else {
		cout << num << " Not Founded\n";
	}

	

}
