11- Write a program that Check if 
the number is prime or not.
                                                 // Answer 
#include <iostream>;
using namespace std;
int main() {
	int num = 0;
	int counter = 0;

	cout << " Please Enter Number \n";
	cin >> num;
	for (int i = 1; i <= num ; i++) {
	
		if (num / i) {
			counter++;
		
		}
	}
	if (counter == 2) {
		cout << " Number Is Prime = " << num;
	}
	else {
		cout << " Number Is Not Prime = " << num;
	}

}

	}
	cout << "Number Of digits is = " << counter;

13- Write a program that Reverse 
an integer.
                                                    // Answer 
#include <iostream>;
using namespace std;
int main() {
	int num {}, rev = 0;
	cout << "Enter Number \n";
	cin >> num;
	for (; num > 0; ) {
		rev = rev * 10 + num % 10 ;
		num /= 10;
	}
	cout << " Reverse = " << rev;

}
